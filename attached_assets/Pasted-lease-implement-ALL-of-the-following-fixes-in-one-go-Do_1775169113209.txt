# FIX 3 — PROSPECT FINDER: MISSING CONVERT ENDPOINT

The frontend calls POST /api/company-prospects/:id/convert but the endpoint does not exist. Build it now:

```typescript
app.post("/api/company-prospects/:id/convert", authenticateToken, async (req: any, res) => {
  try {
    const prospectId = parseInt(req.params.id);
    const userId = req.user.id;

    const prospect = await db.query.companyProspects.findFirst({
      where: eq(companyProspects.id, prospectId)
    });

    if (!prospect) {
      return res.status(404).json({ message: "Prospect not found" });
    }

    const [newEvent] = await db.insert(events).values({
      name: prospect.companyName + " Event",
      clientName: prospect.contactName || prospect.companyName,
      clientEmail: prospect.email || "",
      clientPhone: prospect.phone || "",
      clientCompany: prospect.companyName,
      eventType: prospect.milestoneType || "corporate",
      guestCount: prospect.guestCount || 0,
      budget: prospect.budgetRange || "0",
      currency: "GBP",
      workflowStatus: "new_intake",
      assignedTo: userId,
      notes: `Converted from Prospect Finder. Source: ${prospect.source || "prospect-finder"}. Notes: ${prospect.notes || ""}`,
      createdAt: new Date(),
      updatedAt: new Date()
    }).returning();

    await db.update(companyProspects)
      .set({ status: "won", updatedAt: new Date() })
      .where(eq(companyProspects.id, prospectId));

    await db.insert(enquiries).values({
      name: prospect.contactName || prospect.companyName,
      email: prospect.email || "",
      phone: prospect.phone || "",
      company: prospect.companyName,
      eventType: prospect.milestoneType || "corporate",
      guestCount: prospect.guestCount || 0,
      budgetRange: prospect.budgetRange || "",
      status: "new_intake",
      eventId: newEvent.id,
      source: "prospect-finder",
      notes: prospect.notes || "",
      createdAt: new Date()
    }).catch(() => {});

    await db.insert(auditLogs).values({
      userId,
      action: "prospect_converted",
      entityType: "event",
      entityId: newEvent.id.toString(),
      details: `Prospect "${prospect.companyName}" converted to event`,
      createdAt: new Date()
    }).catch(() => {});

    return res.json({
      eventId: newEvent.id,
      eventName: newEvent.name,
      message: "Prospect successfully converted to event"
    });

  } catch (error) {
    console.error("Convert prospect error:", error);
    return res.status(500).json({ message: "Failed to convert prospect to event" });
  }
});
```
