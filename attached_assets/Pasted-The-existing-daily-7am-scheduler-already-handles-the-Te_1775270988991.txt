function getSenderProfile(country: string): SenderProfile {
  const profiles = await db.select().from(senderProfiles)
  const match = profiles.find(p => 
    p.country_codes.some(c => 
      c.toLowerCase() === country.toLowerCase()
    )
  )
  return match || profiles.find(p => p.is_default)
}

# MISSING PIECE 5 — EMAIL DRAFTING USING TEMPLATES

When drafting an outreach email for any prospect, replace the current on-demand AI generation with template-based generation:

```typescript
async function draftOutreachEmail(prospect: Prospect, contact: Contact): Promise<Email> {
  // 1. Get correct template
  const countryGroup = getNigeriaCountries().includes(prospect.country) ? 'nigeria' : 'uk'
  const template = await db.select().from(outreachTemplates)
    .where(and(
      eq(outreachTemplates.trigger_type, prospect.trigger_type),
      eq(outreachTemplates.country_group, countryGroup)
    )).limit(1)
  
  // 2. Get correct sender
  const sender = await getSenderProfile(prospect.country)
  
  // 3. Fill placeholders
  let subject = template.subject
    .replace('[Company]', prospect.companyName)
    .replace('[N]', calculateYears(prospect.foundingYear))
    .replace('[city]', prospect.city || 'London')
  
  let body = template.body
    .replace('[FirstName]', contact.firstName)
    .replace('[Company]', prospect.companyName)
    .replace('[N]', calculateYears(prospect.foundingYear))
    .replace('[city]', prospect.city || 'London')
    .replace('[SIGNATURE]', sender.signature_block)
  
  // 4. Use Agent to add ONE personalised sentence only
  // Agent adds a single sentence after the opening paragraph that references
  // something specific about this company (sector detail, recent news, city context)
  // It does NOT rewrite the template — it only adds personalisation
  const personalisationSentence = await callAgent(
    `Add ONE sentence (max 20 words) to personalise this email for ${prospect.companyName}, 
     a ${prospect.industry} company in ${prospect.city}. 
     Their trigger: ${prospect.milestone_flag}. 
     The sentence should feel natural after the opening paragraph.
     Return ONLY the sentence, nothing else.`,
    { task: 'email_draft' }
  )
  
  // Insert personalisation after first paragraph
  const paragraphs = body.split('\n\n')
  paragraphs.splice(1, 0, personalisationSentence)
  body = paragraphs.join('\n\n')
  
  return {
    to: contact.email,
    from: sender.from_email,
    fromName: sender.sender_name,
    subject,
    body,
    prospectId: prospect.id,
    contactId: contact.id,
    templateId: template.id,
    status: 'pending_approval'
  }
}
```

This approach:
- Uses your approved templates as the base (no AI rewriting your words)
- Adds one personalised sentence per company for relevance
- Signs with the correct sender per country
- Saves as pending_approval so staff review before sending
