const TODAY = new Date().toISOString().slice(0, 10);
const REJECTED = ["awarded","closed","cancelled","void","suspended","withdrawn","expired","complete"];

function isTenderValid(t) {
  if (!t.deadline || t.deadline < TODAY) return false;
  if (!t.deadline.match(/^202[6-9]|^203/)) return false;
  if (REJECTED.some(s => t.status?.toLowerCase().includes(s))) return false;
  return true;
}

async function fetchTenders(categoryId, keywords) {
  const response = await fetch("/api/search-tenders", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ category: categoryId, keywords })
  });

  if (!response.ok) throw new Error("Search failed");
  const data = await response.json();

  // Frontend safety filter — second pass
  return data.filter(isTenderValid);
}
